document.getElementById("saveNote").addEventListener("click", () => { const note = document.getElementById("noteInput").value; localStorage.setItem("note", note); document.getElementById("savedNote").innerText = "Saved: " + note; }); window.addEventListener("load", () => { const note = localStorage.getItem("note"); if (note) { document.getElementById("savedNote").innerText = "Saved: " + note; } });
